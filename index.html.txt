    // OAuth 2.0 Web Client ID from Google Cloud - fetched dynamically
    let GOOGLE_CLIENT_ID = '';

    // Fetch Google Client ID from server
    async function loadGoogleClientId() {
        try {
            const response = await fetch('/api/google-client-id');
            const data = await response.json();
            GOOGLE_CLIENT_ID = data.clientId;
            console.log('Google Client ID loaded');
        } catch (error) {
            console.error('Failed to load Google Client ID:', error);
            GOOGLE_CLIENT_ID = '338774598801-1pj8tukietpiupfpt89lucjt17odm2hj.apps.googleusercontent.com'; // fallback
        }
    }

        // Minimal JWT decoder for client-side ID token parsing (no signature verification)
        function decodeJwt (token) {
            try {
                const parts = token.split('.');
                if (parts.length !== 3) return null;
                const payload = parts[1];
                const json = atob(payload.replace(/-/g, '+').replace(/_/g, '/'));
                return JSON.parse(decodeURIComponent(escape(json)));
            } catch (e) {
                return null;
            }
        }
        const { createApp } = Vue;

        createApp({
            data() {
                return {
                    blogTitle: 'My Personal Blog',
                    currentView: 'blog',
                    selectedPost: null,
                    posts: [],
                    categories: [],
                    searchQuery: '',
                    selectedCategory: '',
                    selectedTag: '',
                    selectedArchive: '',
                    isLoggedIn: false,
                    entryKey: '',
                    entryKeyVerified: false,
                    entryKeyError: '',
                    loginUsername: '',
                    loginPassword: '',
                    loginError: '',
                    showReset: false,
                    resetEmail: '',
                    resetMessage: '',
                    showNewPostForm: false,
                    editingPost: null,
                    postForm: {
                        title: '',
                        author: '',
                        content: '',
                        categoryId: '',
                        tags: ''
                    }
                };
            },
            mounted() {
                this.loadPosts();
                this.loadCategories();
                // Initialize Google Identity after mount
                this.initGoogleIdentity();
                // Check server session status
                this.checkAuthStatus();
            },
            computed: {
                filteredPosts() {
                    let filtered = this.posts;

                    // Search filter
                    if (this.searchQuery) {
                        const query = this.searchQuery.toLowerCase();
                        filtered = filtered.filter(post =>
                            post.title.toLowerCase().includes(query) ||
                            post.content.toLowerCase().includes(query) ||
                            post.tags.some(tag => tag.toLowerCase().includes(query))
                        );
                    }

                    // Category filter
                    if (this.selectedCategory) {
                        filtered = filtered.filter(post => post.categoryId === this.selectedCategory);
                    }

                    // Tag filter
                    if (this.selectedTag) {
                        filtered = filtered.filter(post => post.tags.includes(this.selectedTag));
                    }

                    // Archive filter
                    if (this.selectedArchive) {
                        const archiveDate = new Date(this.selectedArchive);
                        filtered = filtered.filter(post => {
                            const postDate = new Date(post.date);
                            return postDate.getFullYear() === archiveDate.getFullYear() &&
                                   postDate.getMonth() === archiveDate.getMonth();
                        });
                    }

                    return filtered;
                },
                allTags() {
                    const tags = new Set();
                    this.posts.forEach(post => {
                        post.tags.forEach(tag => tags.add(tag));
                    });
                    return Array.from(tags).sort();
                },
                archives() {
                    const archiveMap = new Map();
                    this.posts.forEach(post => {
                        const date = new Date(post.date);
                        const key = `${date.getFullYear()}-${String(date.getMonth() + 1).padStart(2, '0')}`;
                        const label = date.toLocaleDateString('en-US', { year: 'numeric', month: 'long' });
                        archiveMap.set(key, label);
                    });
                    return Array.from(archiveMap.entries()).map(([key, label]) => ({ key, label })).sort((a, b) => b.key.localeCompare(a.key));
                },
                allCategories() {
                    return [
                        { id: 'all', name: 'All' },
                        ...this.categories
                    ];
                },
                hasFilters() {
                    return this.searchQuery || this.selectedCategory || this.selectedTag || this.selectedArchive;
                }
            },
            methods: {
                initGoogleIdentity() {
                    // Wait until the Google Identity script has loaded
                    if (!window.google || !window.google.accounts || !window.google.accounts.id) {
                        // Retry shortly if script hasn't loaded yet
                        setTimeout(() => this.initGoogleIdentity(), 200);
                        return;
                    }

                    // Render the Google Sign-In button
                    window.google.accounts.id.initialize({
                        client_id: GOOGLE_CLIENT_ID,
                        callback: (response) => this.handleGoogleCredential(response)
                    });

                    window.google.accounts.id.renderButton(
                        document.getElementById('gSignInButton'),
                        { theme: 'outline', size: 'large', width: '250' } // customization
                    );

                    // Optionally prompt the One Tap UX (commented out by default)
                    // window.google.accounts.id.prompt();
                },

                handleGoogleCredential(response) {
                    // response.credential is a JWT ID token. Send it to server for verification.
                    const token = response && response.credential;
                    if (!token) {
                        this.loginError = 'Google sign-in failed';
                        return;
                    }

                    fetch('/auth/google', {
                        method: 'POST',
                        headers: { 'Content-Type': 'application/json' },
                        body: JSON.stringify({ id_token: token })
                    })
                    .then(r => r.json())
                    .then(data => {
                        if (data && data.success) {
                            this.isLoggedIn = true;
                            this.setData('isLoggedIn', true);
                            this.loginError = '';
                            this.loginUsername = '';
                            this.loginPassword = '';
                        } else {
                            this.loginError = data && data.error ? data.error : 'Server verification failed';
                        }
                    })
                    .catch(err => {
                        console.error(err);
                        this.loginError = 'Network error';
                    });
                },

                async checkAuthStatus() {
                    try {
                        const resp = await fetch('/auth/status');
                        const data = await resp.json();
                        if (data.loggedIn) {
                            this.isLoggedIn = true;
                        } else {
                            this.isLoggedIn = false;
                        }
                    } catch (e) {
                        this.isLoggedIn = false;
                    }
                },
                async loadPosts() {
                    try {
                        const resp = await fetch('/api/posts');
                        const data = await resp.json();
                        this.posts = data.posts || [];
                    } catch (e) {
                        console.error('Failed to load posts:', e);
                        this.posts = [];
                    }
                },
                async loadCategories() {
                    try {
                        const resp = await fetch('/api/categories');
                        const data = await resp.json();
                        this.categories = data.categories || [];
                    } catch (e) {
                        console.error('Failed to load categories:', e);
                        this.categories = [];
                    }
                },
                clearFilters() {
                    this.searchQuery = '';
                    this.selectedCategory = '';
                    this.selectedTag = '';
                    this.selectedArchive = '';
                },
                toggleCategory(categoryId) {
                    if (categoryId === 'all') {
                        this.selectedCategory = '';
                    } else {
                        this.selectedCategory = categoryId;
                    }
                },
                getReadingTime(content) {
                    const wordsPerMinute = 200;
                    const words = content.trim().split(/\s+/).length;
                    const minutes = Math.ceil(words / wordsPerMinute);
                    return `${minutes} min read`;
                },
                setData(key, value) {
                    this[key] = value;
                },
                verifyEntryKey() {
                    if (!this.entryKey) {
                        this.entryKeyError = 'Please enter the entry key';
                        return;
                    }
                    // For demo purposes, use a hardcoded key. In production, this should be server-verified.
                    if (this.entryKey === 'admin123') {
                        this.entryKeyVerified = true;
                        this.entryKeyError = '';
                        this.entryKey = '';
                    } else {
                        this.entryKeyError = 'Invalid entry key';
                    }
                },
                login() {
                    // Try server-side login first
                    fetch('/auth/login', {
                        method: 'POST',
                        headers: { 'Content-Type': 'application/json' },
                        body: JSON.stringify({ username: this.loginUsername, password: this.loginPassword })
                    })
                    .then(r => r.json())
                    .then(data => {
                        if (data && data.success) {
                            this.isLoggedIn = true;
                            this.setData('isLoggedIn', true);
                            this.loginError = '';
                            this.loginUsername = '';
                            this.loginPassword = '';
                        } else {
                            this.loginError = data && data.error ? data.error : 'Invalid username or password';
                        }
                    })
                    .catch(() => this.loginError = 'Network error');
                },

                requestReset() {
                    if (!this.resetEmail) { this.resetMessage = 'Please enter an email'; return; }
                    fetch('/auth/request-reset', {
                        method: 'POST',
                        headers: { 'Content-Type': 'application/json' },
                        body: JSON.stringify({ email: this.resetEmail })
                    })
                    .then(r => r.json())
                    .then(() => {
                        this.resetMessage = 'If an account exists, a reset email has been sent.';
                        this.resetEmail = '';
                    })
                    .catch(() => this.resetMessage = 'Network error');
                },
                logout() {
                    // Call server to destroy session
                    fetch('/auth/logout', { method: 'POST' })
                        .then(r => r.json())
                        .then(() => {
                            this.isLoggedIn = false;
                            this.setData('isLoggedIn', false);
                            this.showNewPostForm = false;
                            this.editingPost = null;
                        })
                        .catch(() => {
                            // Fallback: clear client state even if server call fails
                            this.isLoggedIn = false;
                            this.setData('isLoggedIn', false);
                            this.showNewPostForm = false;
                            this.editingPost = null;
                        });
                },
                viewPost(post) {
                    this.selectedPost = post;
                },
                editPost(post) {
                    this.editingPost = post;
                    this.postForm = {
                        title: post.title,
                        author: post.author,
                        content: post.content,
                        categoryId: post.categoryId || '',
                        tags: post.tags.join(', ')
                    };
                },
                cancelEdit() {
                    this.showNewPostForm = false;
                    this.editingPost = null;
                    this.postForm = { title: '', author: '', content: '', categoryId: '', tags: '' };
                },
                async savePost() {
                    if (!this.postForm.title || !this.postForm.content) {
                        alert('Please fill in title and content');
                        return;
                    }

                    const tags = this.postForm.tags
                        .split(',')
                        .map(t => t.trim())
                        .filter(t => t);

                    const postData = {
                        title: this.postForm.title,
                        author: this.postForm.author || 'Anonymous',
                        content: this.postForm.content,
                        categoryId: this.postForm.categoryId,
                        tags: tags
                    };

                    try {
                        if (this.editingPost) {
                            // Update existing post
                            await fetch(`/api/posts/${this.editingPost.id}`, {
                                method: 'PUT',
                                headers: { 'Content-Type': 'application/json' },
                                body: JSON.stringify(postData)
                            });
                        } else {
                            // Create new post
                            await fetch('/api/posts', {
                                method: 'POST',
                                headers: { 'Content-Type': 'application/json' },
                                body: JSON.stringify(postData)
                            });
                        }
                        await this.loadPosts();
                        this.cancelEdit();
                    } catch (e) {
                        console.error('Failed to save post:', e);
                        alert('Failed to save post');
                    }
                },
                async deletePost(id) {
                    if (confirm('Are you sure you want to delete this post?')) {
                        try {
                            await fetch(`/api/posts/${id}`, {
                                method: 'DELETE'
                            });
                            await this.loadPosts();
                        } catch (e) {
                            console.error('Failed to delete post:', e);
                            alert('Failed to delete post');
                        }
                    }
                },
                formatDate(dateString) {
                    const date = new Date(dateString);
                    return date.toLocaleDateString('en-US', {
                        year: 'numeric',
                        month: 'long',
                        day: 'numeric'
                    });
                },
                getExcerpt(content) {
                    return content.length > 150 ? content.substring(0, 150) + '...' : content;
                }
            }
        }).mount('#app');
    </script>
